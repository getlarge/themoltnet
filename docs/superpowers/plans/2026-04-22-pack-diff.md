# Pack Diff API Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `GET /packs/:id/diff/:otherId` and `GET /packs/by-cid/:cid/diff/by-cid/:otherCid` REST endpoints plus a `packs_diff` MCP tool that compares two context packs within the same diary using a DB-side FULL OUTER JOIN classification query.

**Architecture:** The diff is computed entirely in Postgres via a CTE-based self-join on `context_pack_entries` — only rows that actually differ are returned to the application. The service layer resolves packs, enforces same-diary constraints, checks auth, and classifies rows into four buckets (added/removed/reordered/changed). The MCP tool delegates entirely to the REST API via the generated TS client.

**Tech Stack:** TypeBox, Fastify, Drizzle ORM + raw SQL (`db.execute(sql\`...\``)), `@moltnet/api-client` (TS generated), Go `moltnet-api-client` (oapi-codegen generated), Vitest.

**Worktree:** `.worktrees/feat-issue-471-pack-diff` on branch `feat/issue-471-pack-diff`

```typescript
    async diffPacks(
      packAId: string,
      packBId: string,
    ): Promise<PackDiffRow[]> {
      const db = getExecutor(this.db);
      const result = await db.execute(sql`
        WITH diff AS (
          SELECT
            COALESCE(a.entry_id, b.entry_id)   AS entry_id,
            e.title                             AS title,
            a.rank                              AS old_rank,
            b.rank                              AS new_rank,
            a.entry_cid_snapshot               AS old_cid,
            b.entry_cid_snapshot               AS new_cid,
            a.compression_level                AS old_compression,
            b.compression_level                AS new_compression,
            a.packed_tokens                    AS old_packed_tokens,
            b.packed_tokens                    AS new_packed_tokens,
            CASE
              WHEN a.pack_id IS NULL                                     THEN 'added'
              WHEN b.pack_id IS NULL                                     THEN 'removed'
              WHEN a.entry_cid_snapshot != b.entry_cid_snapshot
                OR a.compression_level  != b.compression_level          THEN 'changed'
              WHEN a.rank != b.rank                                      THEN 'reordered'
              ELSE                                                            'unchanged'
            END AS diff_type
          FROM context_pack_entries a
          FULL OUTER JOIN context_pack_entries b
            ON a.entry_id = b.entry_id
            AND b.pack_id = ${packBId}
          LEFT JOIN diary_entries e
            ON e.id = COALESCE(a.entry_id, b.entry_id)
          WHERE a.pack_id = ${packAId}
             OR b.pack_id = ${packBId}
        )
        SELECT * FROM diff WHERE diff_type != 'unchanged'
      `);
      return (result as unknown as { rows: PackDiffRow[] }).rows;
    },
```

```typescript
  async diffPacks(input: DiffPacksInput): Promise<PackDiffResult> {
    if (!input.packAId && !input.packACid) {
      throw new PackServiceError('packAId or packACid is required', 'validation');
    }
    if (!input.packBId && !input.packBCid) {
      throw new PackServiceError('packBId or packBCid is required', 'validation');
    }

    const [packA, packB] = await Promise.all([
      input.packAId
        ? this.deps.contextPackRepository.findById(input.packAId)
        : this.deps.contextPackRepository.findByCid(input.packACid!),
      input.packBId
        ? this.deps.contextPackRepository.findById(input.packBId)
        : this.deps.contextPackRepository.findByCid(input.packBCid!),
    ]);

    if (!packA) throw new PackServiceError('Pack A not found', 'not_found');
    if (!packB) throw new PackServiceError('Pack B not found', 'not_found');

    if (packA.diaryId !== packB.diaryId) {
      throw new PackServiceError(
        'Packs must belong to the same diary',
        'validation',
      );
    }

    await this.assertCanReadPack(packA.id, input.actor);

    // CID short-circuit: identical content, no DB query needed
    if (packA.packCid === packB.packCid) {
      return emptyDiff(packA, packB);
    }

    const rows = await this.deps.contextPackRepository.diffPacks(
      packA.id,
      packB.id,
    );

    return classifyDiffRows(rows, packA, packB);
```

```typescript
function getPackTotalTokens(pack: ContextPackWithCreator): number | null {
  const payload = pack.payload as
    | { compileStats?: { totalTokens?: number } }
    | null
    | undefined;
  return payload?.compileStats?.totalTokens ?? null;
}

function packMeta(pack: ContextPackWithCreator) {
  return {
    id: pack.id,
    packCid: pack.packCid,
    totalTokens: getPackTotalTokens(pack),
    packType: pack.packType,
    createdAt: pack.createdAt,
  };
}

function emptyDiff(
  packA: ContextPackWithCreator,
  packB: ContextPackWithCreator,
): PackDiffResult {
  return {
    added: [],
    removed: [],
    reordered: [],
    changed: [],
    stats: {
      addedCount: 0,
      removedCount: 0,
      reorderedCount: 0,
      changedCount: 0,
      tokenDelta: 0,
      packA: packMeta(packA),
      packB: packMeta(packB),
    },
  };
}

function classifyDiffRows(
  rows: PackDiffRow[],
  packA: ContextPackWithCreator,
  packB: ContextPackWithCreator,
): PackDiffResult {
  const result: PackDiffResult = {
    added: [],
    removed: [],
    reordered: [],
    changed: [],
    stats: {
      addedCount: 0,
      removedCount: 0,
      reorderedCount: 0,
      changedCount: 0,
      tokenDelta: 0,
      packA: packMeta(packA),
      packB: packMeta(packB),
    },
  };

  for (const row of rows) {
    switch (row.diff_type) {
      case 'added':
        result.added.push({
          entryId: row.entry_id,
          title: row.title,
          rank: row.new_rank!,
          entryCidSnapshot: row.new_cid!,
          compressionLevel: row.new_compression!,
          packedTokens: row.new_packed_tokens,
        });
        break;
      case 'removed':
        result.removed.push({
          entryId: row.entry_id,
          title: row.title,
          rank: row.old_rank!,
          entryCidSnapshot: row.old_cid!,
          compressionLevel: row.old_compression!,
          packedTokens: row.old_packed_tokens,
        });
        break;
      case 'reordered':
        result.reordered.push({
          entryId: row.entry_id,
          title: row.title,
          oldRank: row.old_rank!,
          newRank: row.new_rank!,
          entryCidSnapshot: row.new_cid!,
          compressionLevel: row.new_compression!,
          packedTokens: row.new_packed_tokens,
        });
        break;
      case 'changed':
        result.changed.push({
          entryId: row.entry_id,
          rank: row.new_rank ?? row.old_rank!,
          title: row.title,
          oldEntryCidSnapshot: row.old_cid!,
          newEntryCidSnapshot: row.new_cid!,
          oldCompressionLevel: row.old_compression!,
          newCompressionLevel: row.new_compression!,
          oldPackedTokens: row.old_packed_tokens,
          newPackedTokens: row.new_packed_tokens,
          tokenDelta: (row.new_packed_tokens ?? 0) - (row.old_packed_tokens ?? 0),
        });
        break;
    }
  }

  result.stats.addedCount = result.added.length;
  result.stats.removedCount = result.removed.length;
  result.stats.reorderedCount = result.reordered.length;
  result.stats.changedCount = result.changed.length;

  const tokensA = getPackTotalTokens(packA) ?? 0;
  const tokensB = getPackTotalTokens(packB) ?? 0;
  result.stats.tokenDelta = tokensB - tokensA;

  return result;
}
```

```typescript
export async function handlePacksDiff(
  args: PackDiffInput,
  deps: McpDeps,
  context: HandlerContext,
): Promise<CallToolResult> {
  deps.logger.debug({ tool: 'packs_diff' }, 'tool.invoked');
  const token = getTokenFromContext(context);
  if (!token) return errorResult('Not authenticated');

  const packId =
    typeof args.pack_id === 'string' && args.pack_id.trim()
      ? args.pack_id
      : undefined;
  const packCid =
    typeof args.pack_cid === 'string' && args.pack_cid.trim()
      ? args.pack_cid
      : undefined;
  const otherPackId =
    typeof args.other_pack_id === 'string' && args.other_pack_id.trim()
      ? args.other_pack_id
      : undefined;
  const otherPackCid =
    typeof args.other_pack_cid === 'string' && args.other_pack_cid.trim()
      ? args.other_pack_cid
      : undefined;

  if (!packId && !packCid) {
    return errorResult('Exactly one of pack_id or pack_cid must be provided');
  }
  if (packId && packCid) {
    return errorResult('Exactly one of pack_id or pack_cid must be provided');
  }
  if (!otherPackId && !otherPackCid) {
    return errorResult(
      'Exactly one of other_pack_id or other_pack_cid must be provided',
    );
  }
  if (otherPackId && otherPackCid) {
    return errorResult(
      'Exactly one of other_pack_id or other_pack_cid must be provided',
    );
  }

  // ID-based path (both IDs provided)
  if (packId && otherPackId) {
    const { data, error } = await diffContextPacksById({
      client: deps.client,
      auth: () => token,
      path: { id: packId, otherId: otherPackId },
    });
    if (error || !data) {
      deps.logger.error({ tool: 'packs_diff', err: error }, 'tool.error');
      return errorResult(extractApiErrorMessage(error, 'Failed to diff packs'));
    }
    return structuredResult(data);
  }

  // CID-based path (both CIDs provided)
  if (packCid && otherPackCid) {
    const { data, error } = await diffContextPacksByCid({
      client: deps.client,
      auth: () => token,
      path: { cid: packCid, otherCid: otherPackCid },
    });
    if (error || !data) {
      deps.logger.error({ tool: 'packs_diff', err: error }, 'tool.error');
      return errorResult(extractApiErrorMessage(error, 'Failed to diff packs'));
    }
    return structuredResult(data);
  }

  return errorResult(
    'Mixed ID/CID inputs are not supported. Use both IDs or both CIDs.',
  );
}
```
